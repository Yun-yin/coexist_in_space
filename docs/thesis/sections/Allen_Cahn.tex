\section{Allen--Cahn Equation}

Here we consider the Allen--Cahn equation \citep{allen_coherent_1975}
\begin{equation*}
    \frac{\partial u}{\partial t}
    =
    \frac{\partial^2 u}{\partial x^2}
    +
    u(1-u)(u-a).
\end{equation*}
Assuming a travelling wave solution
\begin{equation*}
u(x,t)=u(z),
\qquad
z=x-ct,
\end{equation*}
we obtain the ordinary differential equation
\begin{equation*}
    \frac{\mathrm{d}^2u}{\mathrm{d}z^2}
    +
    c\frac{\mathrm{d}u}{\mathrm{d}z}
    +
    u(1-u)(u-a)
    =
    0.
\end{equation*}
Introducing
\begin{equation*}
p=\frac{\mathrm{d}u}{\mathrm{d}z},
\end{equation*}
the equation can be rewritten as the first-order system
\begin{equation*}
\begin{cases}
\dot{u}=p,\\
\dot{p}=-cp-u(1-u)(u-a),
\end{cases}
\end{equation*}
where the dot denotes differentiation with respect to $z$.

The Jacobian matrix is
\begin{equation*}
\mathbf{J}
=
\begin{pmatrix}
0 & 1\\
3u^2-2(1+a)u+a & -c
\end{pmatrix}.
\end{equation*}

The equilibria (Figure~\ref{fig:fisher_allen}B) are
\begin{equation*}
(u^*,p^*)=(0,0),\qquad (1,0),\qquad (a,0).
\end{equation*}

At $(0,0)$,
\begin{equation*}
\mathbf{J}|_{(0,0)}
=
\begin{pmatrix}
0 & 1\\
a & -c
\end{pmatrix}.
\end{equation*}
Since
\begin{equation*}
\mathrm{tr}(\mathbf{J})=-c<0,
\qquad
\det(\mathbf{J})=-a<0,
\end{equation*}
the equilibrium $(0,0)$ is a saddle point.

At $(1,0)$,
\begin{equation*}
\mathbf{J}|_{(1,0)}
=
\begin{pmatrix}
0 & 1\\
1-a & -c
\end{pmatrix}.
\end{equation*}
Similarly,
\begin{equation*}
\mathrm{tr}(\mathbf{J})=-c<0,
\qquad
\det(\mathbf{J})=-(1-a)<0,
\end{equation*}
so $(1,0)$ is also a saddle point.

At $(a,0)$,
\begin{equation*}
\mathbf{J}|_{(a,0)}
=
\begin{pmatrix}
0 & 1\\
a(a-1) & -c
\end{pmatrix}.
\end{equation*}
In this case,
\begin{equation*}
\mathrm{tr}(\mathbf{J})=-c<0,
\qquad
\det(\mathbf{J})=a(1-a)>0.
\end{equation*}
Therefore, $(a,0)$ is stable. Its characteristic equation is
\begin{equation*}
\lambda^2+c\lambda+a(1-a)=0,
\end{equation*}
with eigenvalues
\begin{equation*}
\lambda
=
\frac{-c\pm\sqrt{c^2-4a(1-a)}}{2}.
\end{equation*}
Depending on the sign of
\begin{equation*}
\Delta=c^2-4a(1-a),
\end{equation*}
the equilibrium may be either a stable node or a spiral sink.

The travelling wave solution corresponds to a heteroclinic trajectory connecting the two saddle points $(1,0)$ and $(0,0)$ (Figure~\ref{fig:trajectory}). Such a connection only exists for a particular wave speed $c$.

\subsection*{Symmetric Case: $a=\frac12$}

When
\begin{equation*}
a=\frac12,
\end{equation*}
the system becomes symmetric with respect to the two stable states $u=0$ and $u=1$. In this case, the travelling wave speed satisfies
\begin{equation*}
c=0,
\end{equation*}
and the travelling wave becomes a stationary interface.

The equation reduces to
\begin{equation*}
\frac{\mathrm{d}^2u}{\mathrm{d}z^2}
=
u(1-u)\left(\frac12-u\right).
\end{equation*}
Multiplying both sides by $\frac{\mathrm{d}u}{\mathrm{d}z}$ gives
\begin{align*}
\frac{\mathrm{d}^2u}{\mathrm{d}z^2}
\frac{\mathrm{d}u}{\mathrm{d}z}
&=
u(1-u)\left(\frac12-u\right)
\frac{\mathrm{d}u}{\mathrm{d}z},
\\
\frac12
\frac{\mathrm{d}}{\mathrm{d}z}
\left(
\frac{\mathrm{d}u}{\mathrm{d}z}
\right)^2
&=
u(1-u)\left(\frac12-u\right)
\frac{\mathrm{d}u}{\mathrm{d}z}.
\end{align*}
Integrating with respect to $z$ yields
\begin{equation*}
\frac12
\left(
\frac{\mathrm{d}u}{\mathrm{d}z}
\right)^2
=
\frac14u^2(1-u)^2.
\end{equation*}
Hence,
\begin{equation*}
\frac{\mathrm{d}u}{\mathrm{d}z}
=
\frac{\sqrt{2}}{2}u(1-u).
\end{equation*}
Separating variables,
\begin{align*}
\frac{1}{u(1-u)}\,\mathrm{d}u
&=
\frac{\sqrt{2}}{2}\,\mathrm{d}z,
\\
\left(
\frac{1}{u}+\frac{1}{1-u}
\right)\mathrm{d}u
&=
\frac{\sqrt{2}}{2}\,\mathrm{d}z.
\end{align*}
After integration,
\begin{equation*}
\ln u - \ln(1-u)
=
\frac{z}{\sqrt{2}}+C.
\end{equation*}
Therefore,
\begin{equation*}
u(z)
=
\frac{1}{1+e^{z/\sqrt{2}+C}}.
\end{equation*}

Choosing the interface center such that
\begin{equation*}
u(0)=\frac12,
\end{equation*}
we obtain $C=0$, and thus (Figure~\ref{fig:fisher_allen}D)
\begin{equation*}
u(z)
=
\frac{1}{1+e^{z/\sqrt{2}}}.
\end{equation*}

\subsection*{Energy Interpretation}

The travelling wave equation can also be studied using an energy method. Consider
\begin{equation*}
\frac{\mathrm{d}^2u}{\mathrm{d}z^2}
+
c\frac{\mathrm{d}u}{\mathrm{d}z}
+
u(1-u)(u-a)
=
0.
\end{equation*}
Multiplying both sides by $\frac{\mathrm{d}u}{\mathrm{d}z}$ gives
\begin{align*}
&
\left(
\frac{\mathrm{d}^2u}{\mathrm{d}z^2}
+
c\frac{\mathrm{d}u}{\mathrm{d}z}
+
u(1-u)(u-a)
\right)
\frac{\mathrm{d}u}{\mathrm{d}z}
=0,
\\
&
\frac12
\frac{\mathrm{d}}{\mathrm{d}z}
\left(
\frac{\mathrm{d}u}{\mathrm{d}z}
\right)^2
+
c
\left(
\frac{\mathrm{d}u}{\mathrm{d}z}
\right)^2
+
u(1-u)(u-a)
\frac{\mathrm{d}u}{\mathrm{d}z}
=
0.
\end{align*}
Define the potential function $W(u)$ by
\begin{equation*}
\frac{\mathrm{d}W}{\mathrm{d}u}
=
u(1-u)(u-a).
\end{equation*}
Integrating,
\begin{equation*}
W(u)
=
-\frac14u^4
+
\frac{1+a}{3}u^3
-
\frac{a}{2}u^2,
\end{equation*}
where we choose $W(0)=0$.
Then
\begin{equation*}
\frac{\mathrm{d}W}{\mathrm{d}z}
=
\frac{\mathrm{d}W}{\mathrm{d}u}
\frac{\mathrm{d}u}{\mathrm{d}z}.
\end{equation*}
Substituting into the previous equation yields
\begin{equation*}
\frac{\mathrm{d}}{\mathrm{d}z}
\left[
\frac12
\left(
\frac{\mathrm{d}u}{\mathrm{d}z}
\right)^2
+
W(u)
\right]
=
-c
\left(
\frac{\mathrm{d}u}{\mathrm{d}z}
\right)^2
\le0.
\end{equation*}
Therefore, the quantity
\begin{equation*}
E(u)
=
\frac12
\left(
\frac{\mathrm{d}u}{\mathrm{d}z}
\right)^2
+
W(u)
\end{equation*}
acts as a decreasing energy functional.
Integrating over the entire domain gives
\begin{equation*}
\left.
\left[
\frac12
\left(
\frac{\mathrm{d}u}{\mathrm{d}z}
\right)^2
+
W(u)
\right]
\right|_{-\infty}^{+\infty}
=
-c
\int_{-\infty}^{+\infty}
\left(
\frac{\mathrm{d}u}{\mathrm{d}z}
\right)^2
\mathrm{d}z.
\end{equation*}

Using the boundary conditions
\begin{equation*}
\lim_{z\to-\infty}u(z)=1,
\qquad
\lim_{z\to+\infty}u(z)=0,
\end{equation*}
and assuming
\begin{equation*}
u'(z)\to0
\qquad
(z\to\pm\infty),
\end{equation*}
we obtain
\begin{equation*}
W(1)-W(0)
=
-c
\int_{-\infty}^{+\infty}
\left(
\frac{\mathrm{d}u}{\mathrm{d}z}
\right)^2
\mathrm{d}z.
\end{equation*}
Hence the wave speed is
\begin{equation*}
c
=
\frac{W(0)-W(1)}
{\displaystyle
\int_{-\infty}^{+\infty}
\left(
\frac{\mathrm{d}u}{\mathrm{d}z}
\right)^2
\mathrm{d}z }.
\end{equation*}

For
\begin{equation*}
a=\frac12,
\end{equation*}
we have
\begin{equation*}
W(0)=W(1),
\end{equation*}
which immediately implies
\begin{equation*}
c=0.
\end{equation*}

\subsection*{General Wave Speed}

Motivated by the stationary solution, assume the travelling wave profile satisfies
\begin{equation*}
\frac{\mathrm{d}u}{\mathrm{d}z}
=
ku(1-u),
\end{equation*}
for some constant $k$.
Then
\begin{align*}
\frac{\mathrm{d}^2u}{\mathrm{d}z^2}
&=
k(1-2u)\frac{\mathrm{d}u}{\mathrm{d}z},
\\
&=
k^2u(1-u)(1-2u).
\end{align*}
Substituting into the travelling wave equation gives
\begin{align*}
&
k^2u(1-u)(1-2u)
+
cku(1-u)
+
u(1-u)(u-a)
=
0.
\end{align*}
Factoring out $u(1-u)$,
\begin{equation*}
u(1-u)
\left[
k^2(1-2u)+ck+u-a
\right]
=
0.
\end{equation*}
Matching coefficients of $u$ and constants yields
\begin{equation*}
2k^2=1,
\qquad
k^2+ck-a=0.
\end{equation*}
Thus,
\begin{equation*}
k=\frac{\sqrt{2}}{2},
\end{equation*}
and
\begin{equation*}
c
=
\sqrt{2}
\left(
a-\frac12
\right).
\end{equation*}
Therefore, the Allen--Cahn equation admits an exact travelling wave solution with speed
\begin{equation*}
c=\sqrt{2}\left(a-\frac12\right),
\end{equation*}
while the wave profile retains the logistic form
\begin{equation*}
u(z)
=
\frac{1}{1+e^{z/\sqrt{2}}}.
\end{equation*}
