    \begin{frame}{Model}
        The framework is a \textbf{diffusion-reaction model}, where the diffusion term is represented by a second-order spatial derivative, and the reaction term follows the GLV dynamics.
        \begin{block}{Diffusion-Reaction Model}
            \begin{equation}
                \begin{aligned}
                    \frac{\partial N_i}{\partial t}
                    = \underbrace{ D_{i}\frac{\partial^2 N_i}{\partial x^2}}_{\text{Dispersal}} + 
                    \underbrace{f(N_j) }_{\text{Interaction}}&\\
                    f(N_j) = & N_i \left( r_i - \sum_{j} A_{i,j} N_j \right)
                \end{aligned}
            \end{equation}
        \end{block}
        
        In short:
        \begin{equation*}
            \partial \mathbf{N} = \mathbf{D} \nabla^2 \mathbf{N} + \mathbf{N} (\mathbf{r} - \mathbf{A}\mathbf{N})
        \end{equation*}
    \end{frame}

\section{$\beta$ Diversity on the Boundary}

    \iffalse
    \begin{frame}
        The previous model provides a null model for species distributions at the boundary between two communities, offering a way to define $\beta$- diversity in a continuous spatial setting. Specifically, for a sequence of observations in which the population of each species changes approximately monotonically, we can first fit a hyperbolic tangent function and then compute the corresponding $\alpha$- and $\gamma$- diversity across the entire one-dimensional space.
    \end{frame}
    \fi

    \begin{frame}
        Normally, at each site, we calculate $\alpha$ from species frequencies $p_i$, and $\gamma$ from the regional frequencies $q_i$.
        \begin{equation*}
            \alpha = -\sum p_i \ln(p_i), \gamma = -\sum q_i \ln(q_i), \beta = \frac{\gamma}{\bar{\alpha}}
        \end{equation*}

        Given the known population distribution pattern, we first fit the abundance distribution for each species, and then use the fitted distributions to estimate site-level and regional diversity components.

        \begin{equation*}
            p_i(x) = \frac{N_i(x)}{\sum_{j}N_j(x)},
            q_i = \frac{\bar{N_i}}{\sum_{j}\bar{N_j}},
            \bar{N_i} = \int_{x} N_i(x)
        \end{equation*}
        \begin{equation*}
            \beta = \frac{\gamma}{\bar{\alpha}} = \frac{\gamma}{\int_{x} \alpha}
        \end{equation*}
    \end{frame}

    \begin{frame}{Example}
        \begin{columns}
            \column{0.6\textwidth}
            \begin{figure}
                \centering
                \includegraphics[scale = 0.45]{six_D.png}
            \end{figure}
            \column{0.4\textwidth}
            $\alpha$:
            \begin{figure}

            \end{figure}
            \begin{align*}
                \bar{\alpha} = 1.181\\
                \gamma = 1.635\\
                \beta = 1.384
            \end{align*}
        \end{columns}
    \end{frame}

    \begin{frame}
        If the stable equilibrium abundance of a species in the first community is $K_1$, and in the second is $K_2$,
        \begin{equation*}
            N(x) = \frac{K_1 - K_2}{1 + e^{-k(x-x_0)}} + K_2
        \end{equation*}

        Community diversity is determined by the shape of the population distribution, which is governed by the parameter $k$, driven by the dispersal rate $D$.

        As a result, this model provides a null expectation for beta diversity, capturing the component driven solely by neutral dispersal. Deviations from this baseline reflect the influence of environmental factors on species turnover.
    \end{frame}
